3 Functions

A function that converts a column col of a data frame x into a list-column (assuming that all the other columns have unique values across rows):

col2listcol <- function(x, col) {
  names_x <- names(x)
  x[1, setdiff(names_x, col)] |> 
    bind_cols(tibble(a = list(x[[col]]))) |> 
    setNames(names_x)
}

6.1 Preparing the data

Generating the presence/absence of K. pneumoniae:

K_pneumoniae_raw <- MALDITOF |> 
  group_by(USUBJID, SampleSchedule) |> 
  group_modify(~ col2listcol(.x, "Identification_MALDITOF")) |> 
  ungroup() |> 
  mutate(Kp = map_lgl(Identification_MALDITOF, ~ "Klebsiella pneumoniae" %in% .x)) |> 
  select(- Identification_MALDITOF) |> 
  arrange(USUBJID, SampleSchedule)

(Kp_msm <- dates |> 
  mutate(DISCHARGE = as.numeric(DISCHARGE - ADMISSION) / (24 * 60),
         ADMISSION = 0) |> 
  pivot_longer(- USUBJID, names_to = "SampleSchedule", values_to = "days") |> 
  left_join(x = K_pneumoniae, y = _, c("USUBJID", "SampleSchedule")) |> 
  mutate(state = Kp + 1,
         days = map_dbl(days, ~ max(.x, 0))))

6.2.1 Bootstrapped CIs

q_list <- boot.msm(Kp_msm_fit, function(x) qmatrix.msm(x)$estimates, 1000)
